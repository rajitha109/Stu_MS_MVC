###Add GetRank Stored Procedure
CREATE PROCEDURE GetClasses
As
SELECT Student.Stuname,Marks.English,Marks.Maths,Marks.Science,Marks.TotalMarks FROM Student INNER JOIN Marks ON Student.Id=Marks.StudentId Order By TotalMarks Desc;
